import { PDFDocument,StandardFonts,rgb } from 'npm:pdf-lib@1.17.1'; import { clean,cors,fail,json,options,rateLimit,requireUser } from '../_shared/core.ts'; function wrap(text:string,max=82){const words=text.split(/\s+/),lines:string[]=[];let line='';for(const word of words){if((line+' '+word).trim().length>max){lines.push(line);line=word}else line=(line+' '+word).trim()}if(line)lines.push(line);return lines} Deno.serve(async req=>{const pre=options(req);if(pre)return pre;try{const user=await requireUser(req);await rateLimit(user.id,'generate-pdf',8);const body=await req.json(),title=clean(body.title,120)||'Beyond The Visa plan',sections=Array.isArray(body.sections)?body.sections.slice(0,30):[];if(!sections.length)return json({error:'At least one section is required.'},400);const pdf=await PDFDocument.create(),regular=await pdf.embedFont(StandardFonts.Helvetica),bold=await pdf.embedFont(StandardFonts.HelveticaBold);let page=pdf.addPage([595,842]),y=790;const line=(text:string,size=10,font=regular,color=rgb(.12,.22,.24))=>{if(y<60){page=pdf.addPage([595,842]);y=790}page.drawText(text,{x:46,y,size,font,color});y-=size+6};line('BEYOND THE VISA',10,bold,rgb(.1,.45,.45));line(title,22,bold);line(`Prepared for ${clean(user.email,100)} · ${new Date().toLocaleDateString('en-GB')}`,9,regular,rgb(.4,.48,.49));y-=12;for(const section of sections){line(clean(section.heading,100),14,bold);for(const row of wrap(clean(section.body,4000))){line(row,10)}y-=8}const bytes=await pdf.save();return new Response(bytes,{headers:{...cors,'Content-Type':'application/pdf','Content-Disposition':'attachment; filename="beyond-the-visa-plan.pdf"','Cache-Control':'no-store'}})}catch(e){return fail(e)}});
